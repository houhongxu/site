function isValid(s: string): boolean {
  const stack = []

  const right2left = {
    ')': '(',
    ']': '[',
    '}': '{',
  }

  for (let i = 0; i < s.length; i++) {
    if (stack.length > 0 && right2left[s[i]] === stack[stack.length - 1]) {
      stack.pop()
    } else {
      stack.push(s[i])
    }
  }

  return !Boolean(stack.length)
}
