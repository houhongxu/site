function sortArray(nums: number[]): number[] {
  const len = nums.length

  if (len <= 1) {
    return nums
  }

  for (let i = 0; i < len - 1; i++) {
    let minIndex = i

    for (let j = i + 1; j < len; j++) {
      if (nums[minIndex] > nums[j]) {
        minIndex = j
      }
    }

    if (minIndex !== i && nums[minIndex] < nums[i]) {
      ;[nums[i], nums[minIndex]] = [nums[minIndex], nums[i]]
    }
  }

  return nums
}
