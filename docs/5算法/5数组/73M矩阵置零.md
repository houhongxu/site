# 73M 矩阵置零

## 解题思路

### 思路一

通过 hash 存一下 0 的行和列

### 代码

```js
/**
 * @param {number[][]} matrix
 * @return {void} Do not return anything, modify matrix in-place instead.
 */
var setZeroes = function (matrix) {
  const hashX = {};
  const hashY = {};

  for (let i = 0; i < matrix.length; i++) {
    for (let j = 0; j < matrix[i].length; j++) {
      if (matrix[i][j] === 0) {
        hashX[i] = true;
        hashY[j] = true;
      }
    }
  }

  for (let i = 0; i < matrix.length; i++) {
    for (let j = 0; j < matrix[i].length; j++) {
      if (hashX[i] || hashY[j]) {
        matrix[i][j] = 0;
      }
    }
  }

  return matrix;
};
```
