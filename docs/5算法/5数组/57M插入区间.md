# 57M 插入区间

## 解题思路

### 思路一

遍历，能合并就合并到 newInterval，不能合并且不是目标位置，就压入 result，通过 newInterval[1] < interval[0]判断目标位置

通过 flag 限制只能插入一次 newInterval

### 代码

```js
/**
 * @param {number[][]} intervals
 * @param {number[]} newInterval
 * @return {number[][]}
 */
var insert = function (intervals, newInterval) {
  let result = [];
  let flag = false;

  for (let i = 0; i < intervals.length; i++) {
    const interval = intervals[i];

    if (
      (newInterval[0] >= interval[0] && newInterval[0] <= interval[1]) ||
      (newInterval[1] >= interval[0] && newInterval[1] <= interval[1]) ||
      (newInterval[0] <= interval[0] && newInterval[1] >= interval[1]) ||
      (newInterval[0] >= interval[0] && newInterval[1] <= interval[1])
    ) {
      // 合并
      newInterval[0] = Math.min(newInterval[0], interval[0]);
      newInterval[1] = Math.max(newInterval[1], interval[1]);
    } else if (newInterval[1] < interval[0] && !flag) {
      // 插入
      result.push(newInterval);
      result.push(interval);
      flag = true;
    } else {
      // 其他
      result.push(interval);
    }
  }

  if (!flag) {
    // []
    result.push(newInterval);
  }

  return result;
};
```
