# 130M 被围绕的区域

## 解题思路

### 思路一

模拟题

根据题意可知，边框开始的 O 都是安全的，其他的 O 都变成 X

先遍历边框，从第一个 O 开始 dfs 查找相邻的四个方向所有 O，返回所有 O 放到 safe 里

然后遍历，非 safe 记录的都为 X

### 代码

```js
/**
 * @param {character[][]} board
 * @return {void} Do not return anything, modify board in-place instead.
 */
var solve = function (board) {
  const m = board.length;
  const n = board[0].length;
  const safe = {};
  const used = [...new Array(m)].map(() => [...new Array(n)]);

  // 查找一整片O
  const findAllO = ([x, y]) => {
    const all = [];

    const dfs = ([x, y]) => {
      if (x < 0 || x >= m || y < 0 || y >= n || board[x][y] === "X") return;

      if (used[x][y]) return;

      used[x][y] = true;

      all.push([x, y]);

      const up = [x - 1, y];
      const down = [x + 1, y];
      const left = [x, y - 1];
      const right = [x, y + 1];
      const directs = [up, down, left, right];

      directs.forEach((direct) => {
        dfs(direct);
      });
    };

    dfs([x, y]);

    return all;
  };

  // 四个边框开始的一片O都是安全的
  for (let i = 0; i < m; i++) {
    for (let j = 0; j < n; j++) {
      if (safe[`${i},${j}`]) {
        continue;
      }

      if (i === 0 || i === m - 1 || j === 0 || j === n - 1) {
        if (board[i][j] === "O") {
          const all = findAllO([i, j]);

          all.forEach((item) => {
            safe[`${item[0]},${item[1]}`] = true;
          });
        }
      }
    }
  }

  // 其他O都改成X
  for (let i = 0; i < m; i++) {
    for (let j = 0; j < n; j++) {
      if (!safe[`${i},${j}`]) {
        board[i][j] = "X";
      }
    }
  }

  return board;
};
```
