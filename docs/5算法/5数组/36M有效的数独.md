# 36M 有效的数独

## 解题思路

### 思路一

二维数组模拟，根据题意做三种判断即可

### 代码

```js
/**
 * @param {character[][]} board
 * @return {boolean}
 */
var isValidSudoku = function (board) {
  for (let i = 0; i < 9; i++) {
    const hash = {};

    for (let j = 0; j < 9; j++) {
      if (board[i][j] !== ".") {
        if (hash[board[i][j]]) {
          return false;
        }

        hash[board[i][j]] = true;
      }
    }
  }

  for (let i = 0; i < 9; i++) {
    const hash = {};

    for (let j = 0; j < 9; j++) {
      if (board[j][i] !== ".") {
        if (hash[board[j][i]]) {
          return false;
        }

        hash[board[j][i]] = true;
      }
    }
  }

  const positions = [
    [0, 0],
    [0, 3],
    [0, 6],
    [3, 0],
    [3, 3],
    [3, 6],
    [6, 0],
    [6, 3],
    [6, 6],
  ];

  for (const [x, y] of positions) {
    const hash = {};

    for (let i = x; i < x + 3; i++) {
      for (let j = y; j < y + 3; j++) {
        if (board[i][j] !== ".") {
          if (hash[board[i][j]]) {
            return false;
          }

          hash[board[i][j]] = true;
        }
      }
    }
  }

  return true;
};
```
