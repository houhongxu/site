# 74M 搜索二维矩阵

## 解题思路

### 思路一

二分查找，行列分别二分查找，先找行，注意行的查找目标是一个范围，列的就是正常的二分查找

### 代码

```js
/**
 * @param {number[][]} matrix
 * @param {number} target
 * @return {boolean}
 */
var searchMatrix = function (matrix, target) {
  const m = matrix.length;
  const n = matrix[0].length;
  let midRow;
  let midCol;

  if (m === 1 && n === 1) {
    return matrix[0][0] === target;
  }

  for (let i = 0, j = m - 1; i <= j; ) {
    midRow = Math.floor((i + j) / 2);

    // 因为在查找指定行时不是精确等于，所以二分查找目标条件也是范围
    if (
      midRow < m - 1 &&
      target >= matrix[midRow][0] &&
      target < matrix[midRow + 1][0]
    ) {
      break;
    }

    if (target > matrix[midRow][0]) {
      i = midRow + 1;
    } else {
      j = midRow - 1;
    }
  }

  for (let i = 0, j = n - 1; i <= j; ) {
    midCol = Math.floor((i + j) / 2);

    if (target === matrix[midRow][midCol]) {
      return true;
    }

    if (target >= matrix[midRow][midCol]) {
      i = midCol + 1;
    } else {
      j = midCol - 1;
    }
  }

  return false;
};
```
